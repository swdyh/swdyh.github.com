var lc = $('left_container');
var rc_w = rc.offsetWidth;
Keybind.add("h", function() {
    if (lc.style.display == 'none') {
        lc.style.display = 'block';
        rc.style.width = rc_w + 'px';
    }
    else {
        lc.style.display = 'none';
        rc.style.width = 'auto';
    }
});
