で、factを継続渡しスタイルで書くとたぶんこんなふう。

function factCPS(n, k) {
    if (n == 0) {
        return k(1)
    }
    else {
        return factCPS(n - 1, function(u) {return k(u * n)})
    }
}

console.logを直でfactCPSに渡すとエラーになったのでlogをつくった。

var log = function(x){console.log(x)}
for (var i = 0; i <= 10; i++) {
    factCPS(i, log)
}
